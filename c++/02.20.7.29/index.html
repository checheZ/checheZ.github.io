委托构造函数：

一个构造函数想要委托另一个构造函数，那么被委托的构造函数应该包含较 大数量的参数，初始化较多的成员变量。而且在委托其他构造函数后，不能 再进行成员列表初始化，而只能在函数体内进行初始化其他成员变量。such as:

    Clock(int newH, int newM, int newS) :
		hour(newH), minute(newM), second(newS) {}
	Clock() :Clock(0, 2, 0) {}//调用有参数函数的构造函数

复制构造函数：

类名::类名(const 类名&对象名)//复制构造函数{函数体}="delete"“delete”

特点：只有单个形参， 且形参是对本类类型对象的引用（常用const修饰）

指示编译器不生成默认复制构造函数 such as:

public:
    Point(int xx = 0, int yy = 0) { x = xx; y = yy; }//构造函数内联
	Point(const Point& p) = delete;
	//指示编译器不生成默认复制构造函数

总结举例:

//5_4.cpp p63举例  难度很高****
#include <iostream>
using namespace std;

class Point {	//Point类定义
public:	//外部接口
	Point(int x = 0, int y = 0) : x(x), y(y) { //构造函数
		//在构造函数中对count累加，所有对象共同维护同一个count
		count++;
	}	
	Point(Point &p) {	//拷贝构造函数
		x = p.x;
		y = p.y;

		count++;
	}
	~Point() {  count--; }
	int getX() { return x; }
	int getY() { return y; }

	void showCount() {		//输出静态数据成员
		cout << "  Object count = " << count << endl;
	}
private:	//私有数据成员
	int x, y;
	static int count;	//静态数据成员声明，用于记录点的个数
};

int Point::count = 0;	//静态数据成员定义和初始化，使用类名限定

int main() {	//主函数
	Point a(4, 5);	//定义对象a，其构造函数回使count增1
	cout << "Point A: " << a.getX() << ", " << a.getY();
	a.showCount();	//输出对象个数

	Point b(a);	//定义对象b，其构造函数回使count增1
	cout << "Point B: " << b.getX() << ", " << b.getY();
	b.showCount();	//输出对象个数

	return 0;
}

类的组合:定义一个新类 可以用已有类

//p66 类的组合程序举例
#include<iostream>
#include<cmath>
using namespace std;
class point {//point 类的定义
private:
	int x, y;
	static int num;
public:
	point(int a, int b) : x(a), y(b) { }
	point(point& p);
	int getx() { return x; }
	int gety() { return y; }

};
point::point(point& p) {//拷贝构造函数的实现
	x = p.x;
	y = p.y;
	cout << "calling the" << num << endl;
}
int point::num = 0;
/// ----------------------
class line {//line 类的定义
private:
	point p1, p2;//point 类的对象p1.p2
	double len;
public:
	line(point xp1, point xp2);
	line(line& l);
	double getlen() { return len; }

};
line::line(point xp1, point xp2) :p1(xp1), p2(xp2) {
	cout << "calling constructor of the line" << endl;
	double x = static_cast<double>(p1.getx() - p2.getx());
	double y = static_cast<double>(p1.gety() - p2.gety());
	len = sqrt(x * x + y * y);
}
line::line(line& l) :p1(l.p1), p2(l.p2) {
	cout << "calling the copy construct of line" << endl;
	len = l.len;
}
int main() {
	point my1(1, 1), my2(4, 5);	//建立point对象
	line lin(my1, my2);	//建立一个line类的对象
	line lin2(lin);	//利用拷贝 构造函数建立一个新的对象
	cout << "The length of the line is:" << lin.getlen() << endl;
	cout << "the length of the line2 is:" << lin2.getlen() << endl;

	return 0;
};
